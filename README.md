## Install

```bash
npm install --save react-modern-chip
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-modern-chip'
import 'react-modern-chip/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
